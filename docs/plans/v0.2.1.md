        // v0.2.1a: Snapshot stats to avoid DB hits during combat
        var weapon = inventory.GetEquippedWeapon(c.Id);
        combatant.WeaponDamageDie = weapon?.DamageDie ?? 4;

    }

public class StatusEffectService : IStatusEffectService
{
    public void ApplyStatus(Combatant target, StatusEffectType type, int duration, int intensity = 1)
    {
        var existing = target.StatusEffects.FirstOrDefault(e => e.Type == type);

        // Logic for Stacking Rules (SPEC-COMBAT-STATUS 9.1)
        if (existing != null)
        {
            if (IsIntensifyType(type)) existing.Stacks += intensity; // e.g. Bleed
            if (IsRefreshType(type)) existing.Duration = duration;   // e.g. Slowed
        }
        else
        {
            target.StatusEffects.Add(new ActiveStatusEffect(type, duration, intensity));
        }
    }

    public void ProcessTurnStart(Combatant combatant)
    {
        // Handle DoTs
        var bleed = combatant.StatusEffects.FirstOrDefault(e => e.Type == StatusEffectType.Bleeding);
        if (bleed != null)
        {
            int damage = _dice.Roll(bleed.Stacks, 6).Total; // 1d6 per stack
            combatant.CurrentHp -= damage;
            // Log event...
        }
    }
}
