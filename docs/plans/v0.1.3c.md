using System.Text;

namespace RuneAndRust.Engine.Services;

public class TextRedactor
{
    public string RedactText(string fullText, int completionPct)
    {
        if (completionPct >= 100) return fullText;
        if (completionPct <= 0) return "[REDACTED]";

        var sb = new StringBuilder();
        var words = fullText.Split(' ');

        // Simple redaction: Show 1 word for every % of completion
        // Real implementation should use the UnlockThresholds dictionary from v0.1.3a

        for (int i = 0; i < words.Length; i++)
        {
            // Pseudo-random masking stability based on completion
            bool visible = (i * 7 + completionPct) % 100 < completionPct;

            if (visible)
                sb.Append(words[i]);
            else
                sb.Append("[grey]████[/]"); // Spectre.Console markup

            sb.Append(" ");
        }

        return sb.ToString();
    }
}
B. Engine Layer (Commands)
File: RuneAndRust.Engine/Commands/JournalCommand.cs
using RuneAndRust.Core.Interfaces;
using RuneAndRust.Engine.Services;

namespace RuneAndRust.Engine.Commands;

public class JournalCommand : ICommand
{
    private readonly IDataCaptureService _dataService;
    private readonly TextRedactor _redactor;
    // Inject GameState, Logger, etc.

    public async Task ExecuteAsync(string[] args)
    {
        var charId = _gameState.CurrentCharacter.Id;
        var entries = await _dataService.GetPlayerCodex(charId);

        // Map Entities to ViewModel
        var vm = new JournalViewModel();
        foreach(var entry in entries)
        {
            var pct = await _dataService.GetCompletionPercentageAsync(entry.Id, charId);
            var display = new JournalEntryDisplay(
                entry.Title,
                entry.Category.ToString(),
                _redactor.RedactText(entry.FullText, pct),
                pct,
                false // Logic for 'IsNew' can be added later
            );

            // Sort into categories
             if (entry.Category == EntryCategory.Bestiary) vm.BestiaryEntries.Add(display);
             else if (entry.Category == EntryCategory.FieldGuide) vm.FieldGuideEntries.Add(display);
             else vm.CodexEntries.Add(display);
        }

        // Pass VM to UI Renderer (Interface pattern)
        _uiRenderer.RenderJournal(vm);
    }
}

using Spectre.Console;
using RuneAndRust.Engine.ViewModels;

public void RenderJournal(JournalViewModel vm)
{
    AnsiConsole.Clear();
    AnsiConsole.Write(new FigletText("Scavenger's Journal").Color(Color.Cyan1));

    // Create layout
    var layout = new Layout("Root")
        .SplitColumns(
            new Layout("Sidebar").Size(20),
            new Layout("Content")
        );

    // Render Tabs (simplified as a Grid/Table for TUI)
    var table = new Table().Border(TableBorder.Rounded);
    table.AddColumn("Category");
    table.AddColumn("Entries");

    // Add Codex Rows
    var codexMarkup = string.Join("\n", vm.CodexEntries.Select(e => FormatEntryRow(e)));
    table.AddRow("[yellow]Codex[/]", codexMarkup);

    // Add Bestiary Rows
    var beastMarkup = string.Join("\n", vm.BestiaryEntries.Select(e => FormatEntryRow(e)));
    table.AddRow("[red]Bestiary[/]", beastMarkup);

    AnsiConsole.Write(table);

    AnsiConsole.MarkupLine("[grey]Press ANY KEY to close...[/]");
    Console.ReadKey();
}

private string FormatEntryRow(JournalEntryDisplay e)
{
    string statusIcon = e.CompletionPercent >= 100 ? "[green]★[/]" : "[grey]●[/]";
    return $"{statusIcon} {e.Title} ([blue]{e.CompletionPercent}%[/])";
}

3. Decision Tree
	•	Q: Should the Journal stop the game loop?
	◦	A: Yes. In the Terminal phase, the Journal acts as a modal screen (Console.ReadKey() to exit). In the future GUI, it will be a window overlay.
	•	Q: How do we handle "Glitched" text effects?
	◦	A: Redaction Service. Do not store glitched text in the database. Generate it at runtime in TextRedactorbased on the character's current Stress level (inject GameState into TextRedactor to access Stress).
	•	Q: How do players read a specific entry?
	◦	A: Sub-commands. Implement codex <name> or allow selecting an entry ID from the journal list. For v0.1.3c, journal lists summaries, and codex "Name" shows the full redacted text panel.
