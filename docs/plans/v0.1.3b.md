B. Engine Layer (Service Logic)
File: RuneAndRust.Engine/Services/DataCaptureService.cs
using Microsoft.Extensions.Logging;
using RuneAndRust.Core.Entities;
using RuneAndRust.Core.Enums;
using RuneAndRust.Core.Interfaces;
using RuneAndRust.Core.Models;

namespace RuneAndRust.Engine.Services;

public class DataCaptureService : IDataCaptureService
{
    private readonly IDataCaptureRepository _captureRepo;
    private readonly ICodexEntryRepository _entryRepo;
    private readonly ILogger<DataCaptureService> _logger;
    private readonly Random _random = new();

    public DataCaptureService(
        IDataCaptureRepository captureRepo,
        ICodexEntryRepository entryRepo,
        ILogger<DataCaptureService> logger)
    {
        _captureRepo = captureRepo;
        _entryRepo = entryRepo;
        _logger = logger;
    }

    public async Task<DataCapture?> TryGenerateCaptureAsync(LootGenerationContext context, Guid characterId)
    {
        // 1. Base Chance: 15% + Wits Bonus
        int chance = 15 + (context.WitsBonus * 2);
        if (_random.Next(0, 100) > chance) return null;

        // 2. Select Entry Category based on Context
        // e.g., Industrial biome -> Technical or BlightOrigin
        var targetCategory = DetermineCategory(context.BiomeType);

        // 3. Fetch random fragment template (simplified for brevity)
        // In real impl, query DB for entries matching category, pick one, generate fragment
        var fragment = new DataCapture
        {
            CharacterId = characterId,
            Type = CaptureType.TextFragment,
            FragmentContent = "System logs indicate catastrophic failure in Sector 4...",
            Source = $"Salvaged from {context.BiomeType} ruins",
            Quality = 15, // Standard
            DiscoveredAt = DateTime.UtcNow
        };

        // 4. Auto-Assign Logic (90% Confidence threshold simulation)
        // Real logic would match keywords or IDs
        if (_random.NextDouble() > 0.1)
        {
             // Simulate finding the parent entry
             var entry = (await _entryRepo.GetByCategoryAsync(targetCategory)).FirstOrDefault();
             if (entry != null)
             {
                 fragment.CodexEntryId = entry.Id;
                 _logger.LogInformation("Auto-assigned fragment to {EntryTitle}", entry.Title);
             }
        }

        await _captureRepo.AddAsync(fragment);
        await _captureRepo.SaveChangesAsync();

        return fragment;
    }

    private EntryCategory DetermineCategory(BiomeType biome)
    {
        return biome switch
        {
            BiomeType.Industrial => EntryCategory.Technical,
            BiomeType.Organic => EntryCategory.Bestiary,
            _ => EntryCategory.Geography
        };
    }

    // ... Implement AssignFragmentAsync and CalculateEntryCompletionAsync
}

4. Logic Decision Tree
	•	Q: How do we determine fragment quality?
	◦	A: Use the WITS check margin from the InteractionService.
	▪	0-2 Net Successes: Standard Quality (15 Legend).
	▪	3+ Net Successes: Specialist Quality (30 Legend).
	•	Q: Does every enemy drop a Bestiary fragment?
	◦	A: No. Use the 1-3-5 Rule.
	▪	Fragment 1: Common loot/kill.
	▪	Fragment 2: WITS examination or specific loot.
	▪	Fragment 3: Rare drop or Specialist analysis.
	•	Q: How do we handle duplicate fragments?
	◦	A: Conversion. If the logic selects a fragment the player already has, convert it to raw Legend (XP)immediately (e.g., "You find a duplicate record. Gained 5 Legend.") or upgrade the Quality of the existing fragment if the new one is better.
