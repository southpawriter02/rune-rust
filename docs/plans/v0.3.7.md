public void Render(InventoryViewModel vm)
{
    var layout = new Layout("Root")
        .SplitRows(
            new Layout("Header").Size(3),
            new Layout("Content").SplitColumns(
                new Layout("Equipment").Ratio(1),
                new Layout("Backpack").Ratio(2)
            ),
            new Layout("Burden").Size(3)
        );

    // Render Equipment Slots
    var equipGrid = new Grid().AddColumn(new GridColumn().NoWrap());
    foreach(var slot in vm.EquipmentSlots)
    {
        var content = slot.IsEmpty ? "[grey](empty)[/]" : $"[cyan]{slot.ItemName}[/]";
        equipGrid.AddRow($"[bold]{slot.SlotName}:[/] {content}");
    }
    layout["Equipment"].Update(new Panel(equipGrid).Header("Wielded"));

    // Render Backpack Table
    var table = new Table().Expand().Border(TableBorder.None);
    table.AddColumns("Item", "Qty", "Wt", "Val");
    foreach(var item in vm.BackpackItems)
    {
        table.AddRow(item.Name, item.Quantity.ToString(), $"{item.Weight}g", $"{item.Value}₴");
    }
    layout["Backpack"].Update(new Panel(table).Header("Pack"));

    // Render Burden Bar
    layout["Burden"].Update(RenderBurdenBar(vm.CurrentWeight, vm.MaxCapacity, vm.BurdenState));

    AnsiConsole.Write(layout);
}

public Panel RenderRecipeDetails(RecipeView recipe, Inventory inventory)
{
    var grid = new Grid().AddColumns(2);
    grid.AddRow("[bold]Difficulty:[/]", $"{recipe.Difficulty} (WITS)");

    // Ingredients
    grid.AddRow(new Markup("[bold underline]Ingredients[/]"), new Text(""));
    foreach(var ing in recipe.Ingredients)
    {
        bool hasEnough = inventory.Count(ing.Id) >= ing.Quantity;
        string color = hasEnough ? "green" : "red";
        string icon = hasEnough ? "✔" : "✗";
        grid.AddRow($"[{color}]{icon} {ing.Name}[/]", $"x{ing.Quantity}");
    }

    return new Panel(grid).Header("Blueprint");
}

public void RenderEntry(CodexEntryView entry)
{
    var content = new StringBuilder();

    // Title & Metadata
    content.AppendLine($"[bold gold1]{entry.Title}[/]");
    content.AppendLine($"[italic grey]{entry.Category} • {entry.Completeness}% Complete[/]");
    content.AppendLine();

    // Redacted Text
    content.AppendLine(entry.RedactedText);

    // Unlocks
    if (entry.UnlockedKeywords.Any())
    {
        content.AppendLine();
        content.AppendLine("[bold cyan]Insights:[/]");
        foreach(var key in entry.UnlockedKeywords)
            content.AppendLine($" - {key}");
    }

    var panel = new Panel(new Markup(content.ToString())).Expand();
    AnsiConsole.Write(panel);
}
