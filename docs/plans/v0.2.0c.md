This version transitions the combat system from "debug logs" to a playable TUI experience using Spectre.Console.
v0.2.0c: The Interface
Goal: Implement the visual layer for combat, enabling players to perceive the Turn Order, Enemy Status (via narrative cues), and Combat Log in a structured layout.

1. Implementation Workflow
	1	Data Transformation (Engine): Create CombatViewModel to transform raw CombatState into UI-ready data (hiding exact Enemy HP, formatting Turn Order).
	2	Log Buffer (Engine): Implement a rolling CombatLog in the CombatService to store the last N events for display.
	3	Rendering (Terminal): Build CombatScreenRenderer using Spectre.Console Layouts (Header, Main, Log, Footer).
	4	Loop Integration (Terminal): Update TerminalGameLoop to trigger a re-render after every user input during the Combat Phase.

B. Engine Layer (Service Updates)
Update CombatService to maintain a log buffer and generate the ViewModel.
File: RuneAndRust.Engine/Services/CombatService.cs (Partial)
public class CombatService : ICombatService
{
    private readonly Queue<string> _combatLog = new();
    private const int MaxLogHistory = 10;

    // Helper to add formatted logs
    private void LogCombatEvent(string message)
    {
        if (_combatLog.Count >= MaxLogHistory) _combatLog.Dequeue();
        _combatLog.Enqueue(message);
    }

    public CombatViewModel GetViewModel()
    {
        var state = _gameState.CombatState!;
        var player = state.Combatants.First(c => c.IsPlayer);

        return new CombatViewModel(
            state.RoundNumber,
            state.ActiveCombatant.Name,
            state.TurnOrder.Select(MapToView).ToList(),
            _combatLog.ToList(),
            new PlayerStatsView(player.CurrentHp, player.MaxHp, player.CurrentStamina, player.MaxStamina)
        );
    }

    private CombatantView MapToView(Combatant c)
    {
        // Design Pillar: Hide Enemy Numbers
        string hpStatus = c.IsPlayer
            ? $"{c.CurrentHp}/{c.MaxHp}"
            : GetNarrativeHealth(c); // "Healthy", "Wounded", etc.

        return new CombatantView(
            c.Id,
            c.Name,
            c.IsPlayer,
            c.Id == _gameState.CombatState!.ActiveCombatant.Id,
            hpStatus,
            "[ ]", // Placeholder for status effects
            c.Initiative.ToString()
        );
    }
}
C. Terminal Layer (Renderer)
Use Spectre.Console to render the TUI.
File: RuneAndRust.Terminal/Services/CombatScreenRenderer.cs
using Spectre.Console;
using RuneAndRust.Engine.ViewModels;

public class CombatScreenRenderer
{
    public void Render(CombatViewModel vm)
    {
        AnsiConsole.Clear();

        // 1. Header (Player Stats)
        var header = new Grid();
        header.AddColumn();
        header.AddRow(new Markup($"[bold]HP:[/] {FormatHp(vm.PlayerStats)}   [bold]STM:[/] {vm.PlayerStats.CurrentStamina}/{vm.PlayerStats.MaxStamina}"));

        // 2. Turn Order Table
        var table = new Table().Border(TableBorder.Rounded).Expand();
        table.AddColumn("Init");
        table.AddColumn("Name");
        table.AddColumn("Condition");

        foreach (var c in vm.TurnOrder)
        {
            var nameStyle = c.IsActive ? "[bold yellow]>[/] " : "  ";
            var color = c.IsPlayer ? "cyan" : "red";
            table.AddRow(
                c.InitiativeDisplay,
                $"{nameStyle}[{color}]{c.Name}[/]",
                c.HealthStatus
            );
        }

        // 3. Combat Log Panel
        var logContent = string.Join("\n", vm.CombatLog);
        var logPanel = new Panel(new Markup(logContent))
            .Header($"Round {vm.RoundNumber}")
            .Border(BoxBorder.Heavy);

        // 4. Layout Composition
        var layout = new Layout("Root")
            .SplitRows(
                new Layout("Top").SplitColumns(
                    new Layout("Stats").Size(3),
                    new Layout("Main").Ratio(1)),
                new Layout("Log").Size(8)
            );

        layout["Stats"].Update(header);
        layout["Main"].Update(table);
        layout["Log"].Update(logPanel);

        AnsiConsole.Write(layout);
    }

    private string FormatHp(PlayerStatsView p) =>
        $"[{(p.CurrentHp < p.MaxHp / 2 ? "red" : "green")}]{p.CurrentHp}/{p.MaxHp}[/]";
}

4. Logic Decision Tree
	•	Q: Do we render a grid?
	◦	A: No. Per SPEC, we stick to "theater of the mind" for now. Grid rendering is reserved for specific complex encounters (v0.3+).
	•	Q: How do we handle color coding for events?
	◦	A: The CombatService should inject markup tags into the log strings (e.g., [red]Hit[/], [yellow]Crit[/]) before sending to ViewModel.
	•	Q: What happens on window resize?
	◦	A: Spectre.Console handles some flow, but artifacts may occur. We force a re-render on every input loop.
