using RuneAndRust.Core.Enums;
using RuneAndRust.Core.Interfaces;
using RuneAndRust.Core.Models.Combat;

public class AttackResolutionService
{
    private readonly IDiceService _dice;

    public AttackResolutionService(IDiceService dice)
    {
        _dice = dice;
    }

    public AttackResult ResolveMelee(Combatant attacker, Combatant defender)
    {
        // 1. Assemble Pools
        // TODO: Weapon skill integration in v0.2.1
        var attackPool = attacker.GetAttribute(Attribute.Might);

        // 2. Roll (Opposed Check vs Static Defense for now)
        // Note: Full opposed rolls come in v0.2.1. For now, Defense acts as a DC threshold.
        var roll = _dice.Roll(attackPool, "Attack Roll");

        // 3. Calculate Net Successes (Successes - Defense Value)
        // Standard Difficulty 6 (subject to balance)
        int netSuccesses = roll.Successes - (defender.Defense / 2);

        // 4. Determine Outcome
        var outcome = netSuccesses switch
        {
            <= -5 => AttackOutcome.Fumble, // Catastrophic failure
            <= 0 => AttackOutcome.Miss,    // Defender wins ties/miss
            1 or 2 => AttackOutcome.Glancing,
            3 or 4 => AttackOutcome.Solid,
            _ => AttackOutcome.Critical
        };

        // 5. Calculate Raw Damage Base
        // Base Damage = MIGHT + Weapon Die (d6 default for now)
        int rawDamage = attacker.GetAttribute(Attribute.Might) + _dice.RollSingle(6, "Weapon Dmg");

        if (outcome == AttackOutcome.Critical) rawDamage *= 2;
        if (outcome == AttackOutcome.Glancing) rawDamage /= 2;

        return new AttackResult(outcome, netSuccesses, rawDamage, netSuccesses > 0);
    }
}

public string ExecutePlayerAttack(string targetName)
{
    var state = _gameState.CombatState;
    if (state == null || !state.IsPlayerTurn) return "It is not your turn.";

    // 1. Identify Target
    var target = state.TurnOrder.FirstOrDefault(c =>
        !c.IsPlayer && c.Name.Contains(targetName, StringComparison.OrdinalIgnoreCase));

    if (target == null) return "Target not found.";

    // 2. Resolve Logic
    var result = _attackResolution.ResolveMelee(state.ActiveCombatant, target);

    // 3. Apply Consequences
    if (result.IsHit)
    {
        int finalDamage = Math.Max(1, result.RawDamage - target.Soak); // Min 1 rule
        target.CurrentHp -= finalDamage;

        // Check Death
        if (target.CurrentHp <= 0)
        {
            state.TurnOrder.Remove(target);
            return $"CRITICAL! You struck {target.Name} for {finalDamage} damage. They fall dead.";
        }

        return $"Hit! ({result.Outcome}) You dealt {finalDamage} damage to {target.Name}.";
    }

    return $"Miss! Your attack was evaded by {target.Name}.";
}

4. Decision Tree: Logic Flow
	•	Q: Does "Defense" reduce damage or hit chance?
	◦	A: Hit Chance. Defense reduces the attacker's Net Successes. Soak (Armor/Sturdiness) reduces Damage.
	•	Q: What happens on a Tie?
	◦	A: Defender Wins. 0 Net Successes is a Miss.
	•	Q: How do we handle Stamina cost?
	◦	A: In CombatService.ExecutePlayerAttack, deduct Stamina before rolling. If CurrentStamina < Cost, return "Exhausted" error. (Start with flat 25 cost).
