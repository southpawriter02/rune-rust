	▪	HP = Current + (Max - Current) * 0.75.
	▪	Stamina = Max.
	▪	AP = Current + (Max - Current) * 0.5 (Mystics only).
	▪	Stress = Current - (WILL * 5). Min 0.
	•	Resource Logic:
	◦	If Inventory contains Ration + Water -> Consume -> Normal Recovery.
	◦	If Inventory missing supplies -> Apply [Exhausted] status -> Recovery capped at 50%.

public RestResult PerformRest(Character character, RestType type)
{
    // 1. Resource Check
    bool hasSupplies = _inventory.HasItem(character, "Ration") && _inventory.HasItem(character, "Water");

    if (type == RestType.Wilderness)
    {
        if (hasSupplies)
        {
            _inventory.RemoveItem(character, "Ration", 1);
            _inventory.RemoveItem(character, "Water", 1);
        }
        else
        {
            _statusEffects.Apply(character, StatusEffectType.Exhausted, -1); // Indefinite duration
        }
    }

    // 2. Calculate Recovery
    int hpRecovered = 0;
    int stressRecovered = 0;

    if (type == RestType.Sanctuary)
    {
        hpRecovered = character.MaxHp - character.CurrentHp;
        character.CurrentHp = character.MaxHp;
        character.CurrentStamina = character.MaxStamina;
        character.PsychicStress = 0;
        // ... Reset Cooldowns ...
    }
    else // Wilderness
    {
        // ... Implement Wilderness Formulas ...
    }

    // 3. Time Advance
    _timeService.AdvanceTime(8 * 60);

    return new RestResult(hpRecovered, stressRecovered, !hasSupplies);
}

public async Task<AmbushResult> CheckForAmbushAsync(Character character, Room room)
{
    // 1. Camp Craft Check
    var skill = character.GetSkillRank(SkillType.WastelandSurvival);
    var roll = _dice.Roll(character.Attributes[Attribute.Wits] + skill, "Camp Craft");

    // 2. Chance Calculation
    int baseChance = 30 + ((int)room.DangerLevel * 10);
    int finalChance = Math.Max(5, baseChance - (roll.Successes * 5)); // Always 5% min risk

    // 3. Resolution
    if (_random.Next(100) < finalChance)
    {
        var encounter = await _encounterGenerator.GenerateAmbush(room);
        return AmbushResult.Triggered(encounter);
    }

    return AmbushResult.Safe();
}

# Changelog: v0.3.2 - Rest & Recovery
**Release Date:** 2025-XX-XX

## Summary
Introduced the Resting System. Players must now manage exhaustion and supplies. Resting in the wilderness risks ambushes, while finding a Runic Anchor allows for safe, full recovery and progression.

## Features
- **Two Rest Types:**
    - **Camp (Wilderness):** Restores 75% HP, reduces Stress based on WILL. Requires Rations/Water. Risk of Ambush.
    - **Sanctuary (Anchor):** Restores 100% HP/Stamina, clears Stress. Safe. Unlocks leveling (`saga`).
- **Ambush System:** Resting in dangerous biomes triggers a Camp Craft check. Failure results in an immediate combat encounter with enemies having initiative.
- **Resource Management:** Resting consumes 1 Ration and 1 Water. Lack of supplies causes **[Exhausted]** status (halved regen).
- **Time Passing:** Resting advances the world clock by 8 hours.

## Technical
- Implemented `RestService` and `AmbushCalculator`.
- Integrated `WastelandSurvival` skill into camp safety checks.
