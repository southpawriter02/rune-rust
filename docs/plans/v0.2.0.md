B. Engine Layer (Logic)
We need an InitiativeService to handle the sorting logic defined in SPEC-CORE-DICE (d10 + FINESSE + WITS).
File: RuneAndRust.Engine/Services/InitiativeService.cs
using RuneAndRust.Core.Interfaces;
using RuneAndRust.Core.Models.Combat;
using Attribute = RuneAndRust.Core.Enums.Attribute;

public class InitiativeService
{
    private readonly IDiceService _dice;

    public InitiativeService(IDiceService dice)
    {
        _dice = dice;
    }

    public void RollInitiative(Combatant combatant)
    {
        // Formula: d10 + FINESSE + WITS (Vigilance)
        var roll = _dice.Roll(1, "Initiative").Successes; // Simplified for v0.2.0a
        var finesse = combatant.GetAttribute(Attribute.Finesse);
        var wits = combatant.GetAttribute(Attribute.Wits);

        combatant.Initiative = roll + finesse + wits;
    }

    public List<Combatant> SortTurnOrder(IEnumerable<Combatant> combatants)
    {
        // Descending order. Ties broken by Finesse, then Coin flip (random)
        return combatants
            .OrderByDescending(c => c.Initiative)
            .ThenByDescending(c => c.GetAttribute(Attribute.Finesse))
            .ThenBy(c => Guid.NewGuid()) // Random tie breaker
            .ToList();
    }
}
File: RuneAndRust.Engine/Services/CombatService.cs
public class CombatService
{
    private readonly GameState _gameState;
    private readonly InitiativeService _initiative;

    public void StartCombat(Character player /*, List<Enemy> enemies */)
    {
        var combatState = new CombatState();

        // 1. Create Combatants
        var playerCombatant = new Combatant
        {
            Name = player.Name,
            Character = player,
            CurrentHp = player.CurrentHp
        };
        combatState.Combatants.Add(playerCombatant);

        // 2. Roll Initiative
        foreach(var c in combatState.Combatants)
        {
            _initiative.RollInitiative(c);
        }

        // 3. Sort Order
        combatState.Combatants = _initiative.SortTurnOrder(combatState.Combatants);

        // 4. Update Global State
        _gameState.CombatState = combatState;
        _gameState.Phase = GamePhase.Combat;
    }

    public void NextTurn()
    {
        var state = _gameState.CombatState;
        state.ActiveTurnIndex++;

        if (state.ActiveTurnIndex >= state.Combatants.Count)
        {
            state.ActiveTurnIndex = 0;
            state.RoundNumber++;
        }
    }
}
