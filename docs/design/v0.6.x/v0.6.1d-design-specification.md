## 1. Executive Summary

### Purpose

The Inventory Grid provides a visual grid-based display of the player's inventory items. Each item occupies a grid slot showing its icon (single character), name abbreviation, and quantity. The grid supports keyboard navigation for item selection and displays item details when a slot is highlighted. Equipment slots are shown separately with clear visual distinction.

### Key Deliverables

| Category | Items |
|----------|-------|
| **Components** | InventoryGrid |
| **Value Objects** | InventorySlot |
| **Interfaces** | IInventoryRenderer |
| **Tests** | ~10 new unit tests |

### Architectural Significance

This version establishes the **Grid Selection Pattern** that will be used for inventory management, equipment, and merchant interactions:
- Grid-based item layout
- Keyboard navigation with cell selection
- Detail panel for selected item
- Visual slot state indicators

/// <summary>
/// Renders a grid-based inventory display with keyboard navigation.
/// </summary>
public class InventoryGrid : IInventoryRenderer
{
    private readonly ScreenLayout _layout;
    private readonly ITerminalService _terminal;
    private readonly IColorThemeService _colorService;
    private readonly ILogger<InventoryGrid> _logger;
    
    private int _selectedRow = 0;
    private int _selectedColumn = 0;
    private int _columns = 6;
    private bool _showEquipment = true;
    
    // Slot dimensions
    private const int SlotWidth = 9;
    private const int SlotHeight = 3;
    private const int EquipmentSlotWidth = 10;
    
    // Symbols
    private const char EmptySlotSymbol = 'Â·';
    
    public InventoryGrid(
        ScreenLayout layout,
        ITerminalService terminal,
        IColorThemeService colorService,
        ILogger<InventoryGrid> logger)
    {
        _layout = layout;
        _terminal = terminal;
        _colorService = colorService;
        _logger = logger;
    }
    
    /// <summary>
    /// Gets the currently selected item, if any.
    /// </summary>
    /// <param name="inventory">The inventory to check.</param>
    /// <returns>The selected item, or null if empty slot selected.</returns>
    public Item? GetSelectedItem(Inventory inventory)
    {
        var index = _selectedRow * _columns + _selectedColumn;
        var items = inventory.Items.ToList();
        return index < items.Count ? items[index] : null;
    }
    
    /// <summary>
    /// Renders the complete inventory view.
    /// </summary>
    /// <param name="inventory">The player's inventory.</param>
    public void RenderInventory(Inventory inventory)
    {
        var panel = _layout.GetPanel(PanelPosition.MainContent);
        if (panel == null)
        {
            _logger.LogWarning("MainContent panel not found for inventory");
            return;
        }
        
        var area = panel.Value.ContentArea;
        _columns = Math.Max(1, area.Width / SlotWidth);
        
        var currentY = area.Y;
        
        // Title with capacity
        RenderTitle(inventory, area.X, currentY, area.Width);
        currentY += 2;
        
        // Equipment slots
        if (_showEquipment)
        {
            RenderEquipmentSlots(inventory.Equipment, area.X, currentY, area.Width);
            currentY += SlotHeight + 2;
        }
        
        // Inventory grid
        RenderGrid(inventory, area.X, currentY, area.Width, area.Height - (currentY - area.Y));
        
        // Item details for selected
        var selectedItem = GetSelectedItem(inventory);
        if (selectedItem != null)
        {
            RenderItemDetails(selectedItem, area);
        }
        
        _logger.LogDebug("Inventory rendered: {ItemCount}/{Capacity} items, " +
            "selected: ({Row}, {Col})", 
            inventory.Items.Count(), inventory.Capacity, 
            _selectedRow, _selectedColumn);
    }
    
    /// <summary>
    /// Renders the equipment slots section.
    /// </summary>
    /// <param name="equipment">The player's equipment.</param>
    /// <param name="x">Start X position.</param>
    /// <param name="y">Start Y position.</param>
    /// <param name="width">Available width.</param>
    public void RenderEquipmentSlots(Equipment equipment, int x, int y, int width)
    {
        var slots = new[]
        {
            (Name: "HEAD", Item: equipment.Head),
            (Name: "CHEST", Item: equipment.Chest),
            (Name: "WEAPON", Item: equipment.Weapon),
            (Name: "SHIELD", Item: equipment.Shield),
            (Name: "HANDS", Item: equipment.Hands),
            (Name: "FEET", Item: equipment.Feet)
        };
        
        var visibleSlots = Math.Min(slots.Length, width / EquipmentSlotWidth);
        var totalWidth = visibleSlots * EquipmentSlotWidth;
        var startX = x + (width - totalWidth) / 2;
        
        for (var i = 0; i < visibleSlots; i++)
        {
            var slot = slots[i];
            var slotX = startX + i * EquipmentSlotWidth;
            
            RenderEquipmentSlot(slot.Name, slot.Item, slotX, y);
        }
    }
    
    /// <summary>
    /// Moves the selection in the grid.
    /// </summary>
    /// <param name="direction">Direction to move.</param>
    /// <param name="inventory">Current inventory for bounds checking.</param>
    public void SelectSlot(Direction direction, Inventory inventory)
    {
        var rows = (int)Math.Ceiling((double)inventory.Capacity / _columns);
        
        switch (direction)
        {
            case Direction.North:
                _selectedRow = Math.Max(0, _selectedRow - 1);
                break;
            case Direction.South:
                _selectedRow = Math.Min(rows - 1, _selectedRow + 1);
                break;
            case Direction.West:
                _selectedColumn = Math.Max(0, _selectedColumn - 1);
                break;
            case Direction.East:
                _selectedColumn = Math.Min(_columns - 1, _selectedColumn + 1);
                break;
        }
        
        _logger.LogDebug("Selection moved to ({Row}, {Col})", _selectedRow, _selectedColumn);
    }
    
    /// <summary>
    /// Handles keyboard input for the inventory grid.
    /// </summary>
    /// <param name="key">The key pressed.</param>
    /// <param name="inventory">Current inventory.</param>
    /// <returns>The action to perform, if any.</returns>
    public InventoryAction? HandleInput(ConsoleKey key, Inventory inventory)
    {
        switch (key)
        {
            case ConsoleKey.UpArrow:
                SelectSlot(Direction.North, inventory);
                return null;
            case ConsoleKey.DownArrow:
                SelectSlot(Direction.South, inventory);
                return null;
            case ConsoleKey.LeftArrow:
                SelectSlot(Direction.West, inventory);
                return null;
            case ConsoleKey.RightArrow:
                SelectSlot(Direction.East, inventory);
                return null;
            case ConsoleKey.Enter:
            case ConsoleKey.U:
                return InventoryAction.Use;
            case ConsoleKey.E:
                return InventoryAction.Equip;
            case ConsoleKey.D:
                return InventoryAction.Drop;
            case ConsoleKey.Escape:
                return InventoryAction.Close;
            default:
                return null;
        }
    }
    
    private void RenderTitle(Inventory inventory, int x, int y, int width)
    {
        var title = $"â•â•â• INVENTORY ({inventory.Items.Count()}/{inventory.Capacity}) â•â•â•";
        var padding = (width - title.Length) / 2;
        
        _terminal.SetCursorPosition(x + padding, y);
        _terminal.Write(title);
    }
    
    private void RenderGrid(Inventory inventory, int x, int y, int width, int height)
    {
        var items = inventory.Items.ToList();
        var rows = (int)Math.Ceiling((double)inventory.Capacity / _columns);
        var visibleRows = Math.Min(rows, height / SlotHeight);
        
        for (var row = 0; row < visibleRows; row++)
        {
            for (var col = 0; col < _columns; col++)
            {
                var index = row * _columns + col;
                var slotX = x + col * SlotWidth;
                var slotY = y + row * SlotHeight;
                var isSelected = row == _selectedRow && col == _selectedColumn;
                
                if (index < items.Count)
                {
                    RenderSlot(items[index], slotX, slotY, isSelected);
                }
                else if (index < inventory.Capacity)
                {
                    RenderEmptySlot(slotX, slotY, isSelected);
                }
            }
        }
    }
    
    private void RenderSlot(Item item, int x, int y, bool isSelected)
    {
        var (topLeft, topRight, bottomLeft, bottomRight, horizontal, vertical) = 
            isSelected 
                ? ('â•”', 'â•—', 'â•š', 'â•', 'â•', 'â•‘')
                : ('â”Œ', 'â”', 'â””', 'â”˜', 'â”€', 'â”‚');
        
        var borderColor = isSelected ? ConsoleColor.Yellow : ConsoleColor.Gray;
        
        // Top border
        _terminal.SetCursorPosition(x, y);
        WriteColored($"{topLeft}{new string(horizontal, SlotWidth - 2)}{topRight}", borderColor);
        
        // Middle with icon and abbreviation
        _terminal.SetCursorPosition(x, y + 1);
        WriteColored(vertical.ToString(), borderColor);
        
        var icon = item.Icon;
        var abbrev = GetAbbreviation(item.Name);
        var content = $" {icon} {abbrev} ";
        _terminal.Write(content.PadRight(SlotWidth - 2));
        
        WriteColored(vertical.ToString(), borderColor);
        
        // Bottom with quantity
        _terminal.SetCursorPosition(x, y + 2);
        WriteColored(vertical.ToString(), borderColor);
        
        var qty = item.Quantity > 1 ? $"x{item.Quantity}" : "";
        var qtyContent = qty.PadLeft(SlotWidth - 2);
        _terminal.Write(qtyContent);
        
        WriteColored(vertical.ToString(), borderColor);
        
        // Bottom border
        _terminal.SetCursorPosition(x, y + 3);
        WriteColored($"{bottomLeft}{new string(horizontal, SlotWidth - 2)}{bottomRight}", borderColor);
    }
    
    private void RenderEmptySlot(int x, int y, bool isSelected)
    {
        var (topLeft, topRight, bottomLeft, bottomRight, horizontal, vertical) = 
            isSelected 
                ? ('â•”', 'â•—', 'â•š', 'â•', 'â•', 'â•‘')
                : ('â”Œ', 'â”', 'â””', 'â”˜', 'â”€', 'â”‚');

        // Top border
        _terminal.SetCursorPosition(x, y);

        // Bottom empty
        _terminal.SetCursorPosition(x, y + 2);
        WriteColored(vertical.ToString(), borderColor);
        _terminal.Write(new string(' ', SlotWidth - 2));
        WriteColored(vertical.ToString(), borderColor);
        
        // Bottom border
        _terminal.SetCursorPosition(x, y + 3);
        WriteColored($"{bottomLeft}{new string(horizontal, SlotWidth - 2)}{bottomRight}", borderColor);
    }
    
    private void RenderEquipmentSlot(string slotName, Item? item, int x, int y)
    {
        // Top with slot name
        _terminal.SetCursorPosition(x, y);
        _terminal.Write($"â”Œ{slotName.PadLeft(4).PadRight(8)}â”");
        
        // Middle with item or empty
        _terminal.SetCursorPosition(x, y + 1);
        if (item != null)
        {
            var icon = item.Icon;
            _terminal.Write($"â”‚  [{icon}]   â”‚");
        }
        else
        {
            _terminal.Write("â”‚  [ ]   â”‚");
        }
        
        // Bottom with item name or empty
        _terminal.SetCursorPosition(x, y + 2);
        var name = item?.Name ?? "empty";
        var displayName = name.Length > 8 ? name[..8] : name.PadLeft((8 + name.Length) / 2).PadRight(8);
        _terminal.Write($"â”‚{displayName}â”‚");
        
        // Bottom border
        _terminal.SetCursorPosition(x, y + 3);
        _terminal.Write("â””â”€â”€â”€â”€â”€â”€â”€â”€â”˜");
    }
    
    private void RenderItemDetails(Item item, (int X, int Y, int Width, int Height) area)
    {
        var detailY = area.Y + area.Height - 7;
        var detailWidth = area.Width - 4;
        var detailX = area.X + 2;
        
        // Border
        _terminal.SetCursorPosition(detailX, detailY);
        _terminal.Write("â”Œ" + new string('â”€', detailWidth - 2) + "â”");
        
        // Item name
        _terminal.SetCursorPosition(detailX, detailY + 1);
        _terminal.Write("â”‚ ");

        _terminal.SetCursorPosition(detailX, detailY + 2);
        var category = $"â”‚ Category: {item.Category}";
        _terminal.Write(category.PadRight(detailWidth - 1) + "â”‚");
        
        // Description
        _terminal.SetCursorPosition(detailX, detailY + 3);
        var desc = item.Description.Length > detailWidth - 4 
            ? item.Description[..(detailWidth - 7)] + "..."
            : item.Description;
        _terminal.Write($"â”‚ {desc}".PadRight(detailWidth - 1) + "â”‚");
        
        // Empty line
        _terminal.SetCursorPosition(detailX, detailY + 4);
        _terminal.Write("â”‚" + new string(' ', detailWidth - 2) + "â”‚");
        
        // Actions
        _terminal.SetCursorPosition(detailX, detailY + 5);
        _terminal.Write("â”‚ [E]quip  [D]rop  [U]se".PadRight(detailWidth - 1) + "â”‚");
        
        // Bottom border
        _terminal.SetCursorPosition(detailX, detailY + 6);
        _terminal.Write("â””" + new string('â”€', detailWidth - 2) + "â”˜");
    }
    
    private static string GetAbbreviation(string name)
    {
        if (name.Length <= 4) return name;
        
        // Try to find meaningful abbreviation
        var words = name.Split(' ');
        if (words.Length > 1)
        {
            return new string(words.Select(w => w[0]).Take(4).ToArray());
        }
        
        return name[..4];
    }
    
    private void WriteColored(string text, ConsoleColor color)
    {
        var prevColor = Console.ForegroundColor;
        Console.ForegroundColor = color;
        _terminal.Write(text);
        Console.ForegroundColor = prevColor;
    }

namespace RuneAndRust.Domain.ValueObjects;

/// <summary>
/// Represents a slot in the inventory grid.
/// </summary>
/// <param name="Row">Grid row position (0-indexed).</param>
/// <param name="Column">Grid column position (0-indexed).</param>
/// <param name="Item">Item in this slot, or null if empty.</param>
/// <param name="IsEquipmentSlot">Whether this is an equipment slot.</param>
/// <param name="EquipmentSlotType">Type of equipment slot, if applicable.</param>
public readonly record struct InventorySlot(
    int Row,
    int Column,
    Item? Item = null,
    bool IsEquipmentSlot = false,
    EquipmentSlotType? EquipmentSlotType = null)
{
    /// <summary>
    /// Gets whether this slot is empty.
    /// </summary>
    public bool IsEmpty => Item == null;
    
    /// <summary>
    /// Gets the linear index of this slot in the grid.
    /// </summary>
    /// <param name="columns">Number of columns in the grid.</param>
    public int GetIndex(int columns) => Row * columns + Column;
    
    /// <summary>
    /// Creates an empty slot at the specified position.
    /// </summary>
    public static InventorySlot Empty(int row, int column) => 
        new(row, column);
    
    /// <summary>
    /// Creates a slot with an item.
    /// </summary>
    public static InventorySlot WithItem(int row, int column, Item item) =>
        new(row, column, item);
}
