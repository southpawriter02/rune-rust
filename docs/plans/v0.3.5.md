public class ExplorationLayout
{
    private Layout _root;

    public ExplorationLayout()
    {
        _root = new Layout("Root")
            .SplitRows(
                new Layout("Header").Size(3),
                new Layout("Body").SplitColumns(
                    new Layout("Main"),
                    new Layout("Sidebar").Size(25)
                ),
                new Layout("Footer").Size(1)
            );
    }

    public void Render(GameState state)
    {
        // Update Header
        _root["Header"].Update(CreateStatusBar(state.Player));

        // Render Full Screen
        AnsiConsole.Write(_root);
    }

    private Panel CreateStatusBar(Character player)
    {
        // Use a Grid for alignment
        var grid = new Grid().AddColumns(3).Width(100);
        grid.AddRow(
            $"[bold]{player.Name}[/]",
            RenderBar("HP", player.CurrentHp, player.MaxHp, Color.Red),
            RenderBar("STA", player.CurrentStamina, player.MaxStamina, Color.Yellow)
        );
        return new Panel(grid).Border(BoxBorder.None);
    }
}

v0.3.5b: The Cartographer (Minimap)
Goal: Implement the MinimapWidget in the sidebar. It displays a 3x3 or 5x5 grid relative to the player's position, handling Z-levels and Fog of War.
1. Architecture & Data Flow
	•	Input: CurrentRoomId from GameState.
	•	Query: NavigationService.GetSurroundingRooms(center, radius).
	•	Rendering:
	◦	Iterate grid (x-1 to x+1, y-1 to y+1).
	◦	Check VisitedRooms list (Fog of War).
	◦	Determine ASCII char based on RoomType (e.g., + for Hall, # for Wall).
2. Logic Decision Trees
	•	Tile Selection Logic:
	1	Is Player Here? -> Render @ (Cyan).
	2	Is Room Visited?
	▪	No: Render (Empty) or ? (if mapped via skill).
	▪	Yes: Check Room Type.
	▪	Corridor: +
	▪	Chamber: O
	▪	Stairs Up/Down: ^ / v
	3	Color:
	▪	Safe: Grey.
	▪	Danger: Red (if known).

public Panel Render(GameState state)
{
    var grid = new Grid();
    for(int i=0; i<3; i++) grid.AddColumn(); // 3x3 Grid

    var center = state.CurrentRoom.Position;

    for (int y = 1; y >= -1; y--) // Render Top-Down
    {
        var row = new List<string>();
        for (int x = -1; x <= 1; x++)
        {
            var targetPos = center.Offset(x, y, 0);
            var room = _mapService.GetRoomAt(targetPos);

            if (x == 0 && y == 0) row.Add("[cyan]@[/]"); // Player
            else if (room == null) row.Add("[grey]·[/]"); // Void
            else if (!state.VisitedRoomIds.Contains(room.Id)) row.Add(" "); // Fog
            else row.Add(GetRoomSymbol(room));
        }
        grid.AddRow(row.ToArray());
    }

    return new Panel(grid).Header("Map");
}

public Panel Render(Room room, List<InteractableObject> objects)
{
    var sb = new StringBuilder();
    sb.AppendLine(room.Description);
    sb.AppendLine();

    if (objects.Any())
    {
        sb.AppendLine("[bold]Visible Objects:[/]");
        foreach(var obj in objects)
        {
            var color = obj.IsContainer ? "gold1" : "grey";
            sb.AppendLine($" - [{color}]{obj.Name}[/]");
        }
    }

    return new Panel(new Markup(sb.ToString()))
        .Header($"[bold orange1]{room.Name}[/]")
        .Expand();
}
