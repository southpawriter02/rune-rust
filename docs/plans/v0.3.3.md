public HazardResult ProcessTrigger(Room room, TriggerType type, object context)
{
    var hazards = _repo.GetActiveHazards(room.Id);
    foreach (var hazard in hazards)
    {
        if (hazard.Trigger == type && ValidateContext(hazard, context))
        {
            // Execute EffectScript using AbilityService's parser
            var result = _abilityService.ExecuteScript(hazard.EffectScript);

            // Update State
            if (hazard.OneTimeUse) hazard.State = HazardState.Destroyed;
            else if (hazard.Cooldown > 0) hazard.State = HazardState.Cooldown;

            return new HazardResult(true, hazard, result);
        }
    }
    return HazardResult.None();
}

public void ProcessRoomTick(GameState state)
{
    var condition = state.CurrentRoom.Condition;
    if (condition == null) return;

    if (!string.IsNullOrEmpty(condition.TurnStartScript))
    {
        foreach (var combatant in state.CombatState.Combatants)
        {
             // Check immunity (e.g. Mechanical vs Poison)
             if (_traitService.IsImmune(combatant, condition)) continue;

             _abilityService.ExecuteScript(condition.TurnStartScript, target: combatant);
        }
    }
}

# Changelog: v0.3.3 - Dynamic Environment
**Release Date:** 2025-XX-XX

## Summary
The dungeon is no longer a static backdrop. Environments now actively shape gameplay through **Dynamic Hazards** (traps, vents, explosives) and **Ambient Conditions** (temperature, psychic pressure). Players must navigate the terrain as carefully as they fight enemies.

## Features
- **Dynamic Hazards:**
    - Interactive objects that trigger on movement or damage.
    - **Steam Vents:** Erupt periodically or when damaged.
    - **Pressure Plates:** Trigger traps when stepped on.
    - **Spore Pods:** Explode into poison clouds when hit.
- **Ambient Conditions:**
    - Room-wide modifiers affecting all combatants.
    - **Psychic Resonance:** +Stress per turn, Magic deal +Damage.
    - **Deep Cold:** Freezing damage per turn, Slows movement.
- **AI Awareness:** Intelligent enemies now attempt to avoid visible hazards.

## Technical
- Implemented `HazardService` and `ConditionService`.
- Integrated `EffectScript` system for environmental effects.
- Updated Dungeon Generation to seed biome-specific hazards.
