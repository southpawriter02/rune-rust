Goal: Close the combat loop. Currently, the player can attack, but enemies are static. This version implements basic Enemy AI, Stamina Regeneration, and the player's Defensive options (Block/Flee).

2. Architecture & Logic Flow
We introduce an automated step in the Turn Cycle.

A. Engine Layer (AI Service)
For v0.2.0d, we use a simple "Aggressive" pattern: If in range, attack.
File: RuneAndRust.Engine/Services/EnemyAIService.cs
public class EnemyAIService
{
    private readonly IAttackResolutionService _attackResolution;

    public EnemyAIService(IAttackResolutionService attackResolution)
    {
        _attackResolution = attackResolution;
    }

    public string DetermineAndExecuteAction(CombatState state)
    {
        var enemy = state.ActiveCombatant;
        var player = state.TurnOrder.First(c => c.IsPlayer);

        // Basic Logic: If we have stamina, attack the player.
        // Enemies use "Standard" attack by default in v0.2.0d
        if (_attackResolution.CanAffordAttack(enemy, AttackType.Standard))
        {
            var result = _attackResolution.ResolveMeleeAttack(enemy, player, AttackType.Standard);

            // Logic to apply damage matches CombatService.ExecutePlayerAttack
            // Note: In a real impl, this logic belongs in a shared "ApplyAttack" method in CombatService
            if (result.IsHit)
            {
                player.CurrentHp -= result.FinalDamage;
                enemy.CurrentStamina -= 25; // Deduct cost
                return $"{enemy.Name} attacks! {result.Outcome} for {result.FinalDamage} damage.";
            }

            enemy.CurrentStamina -= 25;
            return $"{enemy.Name} attacks but misses ({result.Outcome})!";
        }

        // Fallback: Regenerate or Pass
        return $"{enemy.Name} catches their breath (Pass).";
    }
}

public async Task NextTurn()
{
    // ... existing cycling logic ...

    // Regen Stamina at start of turn
    var active = _gameState.CombatState.ActiveCombatant;
    active.CurrentStamina = Math.Min(active.MaxStamina, active.CurrentStamina + 5); // Base regen

    if (!active.IsPlayer)
    {
        // Add a small delay for UX so the player sees the turn change
        await Task.Delay(500);

        var resultMsg = _enemyAI.DetermineAndExecuteAction(_gameState.CombatState);
        LogCombatEvent(resultMsg);

        // Recursively call NextTurn to pass back to player or next enemy
        await NextTurn();
    }
}

public string ExecuteFlee()
{
    var player = _gameState.CombatState.ActiveCombatant;
    // Flee Check: FINESSE + WITS vs Average Enemy FINESSE
    // Simplified for v0.2.0d

    var roll = _dice.Roll(player.GetAttribute(Attribute.Finesse) + player.GetAttribute(Attribute.Wits), "Flee Check");

    if (roll.Successes >= 1) // Basic threshold
    {
        EndCombat();
        return "You escaped!";
    }

    NextTurn();
    return "You failed to escape!";
}
