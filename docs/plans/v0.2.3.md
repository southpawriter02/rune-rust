public bool DeductResource(Combatant combatant, ResourceType type, int amount)
{
    switch (type)
    {
        case ResourceType.Stamina:
            if (combatant.CurrentStamina < amount) return false;
            combatant.CurrentStamina -= amount;
            return true;
        case ResourceType.Aether:
             // Mystics can Overcast (spend HP if AP is empty) - SPEC-CORE-RES-AETHER
            if (combatant.CurrentAp < amount) return HandleOvercast(combatant, amount);
            combatant.CurrentAp -= amount;
            return true;
        default:
            return false;
    }
}

public AbilityResult ExecuteAbility(Combatant user, Combatant target, ActiveAbility ability)
{
    // 1. Validation & Cost
    if (!CanUse(user, ability)) return AbilityResult.Failure("Cannot use.");
    _resourceService.DeductResource(user, ResourceType.Stamina, ability.StaminaCost);

    // 2. Parse Script
    var effects = ability.EffectScript.Split(';');
    foreach (var effect in effects)
    {
        var parts = effect.Split(':');
        switch (parts)
        {
            case "DAMAGE":
                _attackResolution.ApplyAbilityDamage(user, target, parts, parts);
                break;
            case "STATUS":
                var type = Enum.Parse<StatusEffectType>(parts);
                _statusService.ApplyStatus(target, type, int.Parse(parts));
                break;
        }
    }

    // 3. Cooldown
    user.Cooldowns[ability.Id] = ability.CooldownTurns;

    return AbilityResult.Success();
}

# Changelog: v0.2.3 - The Hero's Toolkit
**Release Date:** 2025-XX-XX

## Summary
Introduced the Active Ability system, Resource economy, and Tier 1 class kits. Players can now use specialized moves consuming Stamina or Aether.

## Features
- **Resource System:** Stamina (Physical) and Aether (Magical) now track consumption and regeneration.
- **Ability Engine:** Data-driven ability system supporting damage, healing, and status effect scripts.
- **Tier 1 Kits:**
    - **Warrior:** Wild Swing (Heavy Damage), Defensive Stance.
    - **Skirmisher:** Precise Shot (Ranged).
    - **Mystic:** Aether Dart (Magic Damage).
    - **Adept:** Mend Wound (Healing).
- **Cooldowns:** Abilities now enforce turn-based cooldowns.

## Technical
- Added `IResourceService` and `IAbilityService`.
- Implemented `EffectScript` parser for modular ability logic.
- Updated `Combatant` to track Cooldowns and dynamic Resources.
