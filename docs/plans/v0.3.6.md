public Panel Render(CombatViewModel vm)
{
    var battlefield = new Grid();
    battlefield.AddColumn(); // Player Side
    battlefield.AddColumn(); // VS Separator
    battlefield.AddColumn(); // Enemy Side

    var playerRows = RenderSide(vm.PlayerTeam);
    var enemyRows = RenderSide(vm.EnemyTeam);

    battlefield.AddRow(playerRows, new Markup("[grey]|[/]"), enemyRows);

    return new Panel(battlefield)
        .Header("Tactical View")
        .Border(BoxBorder.Rounded);
}

private Grid RenderSide(IEnumerable<CombatantView> team)
{
    var sideGrid = new Grid().AddColumn();
    // Render Back Row
    var backRow = new Grid().AddColumns(4); // Up to 4 slots
    foreach(var c in team.Where(x => x.Row == RowPosition.Back))
        backRow.AddColumn(RenderCombatantCard(c));

    // Render Front Row
    var frontRow = new Grid().AddColumns(4);
    foreach(var c in team.Where(x => x.Row == RowPosition.Front))
        frontRow.AddColumn(RenderCombatantCard(c));

    sideGrid.AddRow(backRow);
    sideGrid.AddRow(frontRow);
    return sideGrid;
}

public Panel Render(List<CombatantView> turnOrder, int activeIndex)
{
    var table = new Table().Border(TableBorder.None).HideHeaders();

    // Create visual queue
    for(int i = 0; i < 6; i++) table.AddColumn();

    var cells = new List<string>();
    int count = turnOrder.Count;

    for(int i = 0; i < 6; i++)
    {
        var targetIndex = (activeIndex + i) % count;
        var unit = turnOrder[targetIndex];
        var style = i == 0 ? "[bold yellow]" : "[grey]"; // Highlight active
        cells.Add($"{style}[[{unit.Initiative}]] {unit.Name}[/]");
    }

    table.AddRow(cells.ToArray());
    return new Panel(table).Header("Initiative");
}

v0.3.6c: The Telegraph (Intent & Status)
Goal: Implement the "Telegraph" system where enemies reveal their planned actions (Attack, Block, Buff) based on the player's perception stats, and visualize active Status Effects (Bleed, Stun).

	•	Intent System:
	◦	Enemies decide their action at the start of the round (or end of previous).

	◦	CombatService checks player's WITS vs Enemy Deception (or flat check).
	◦	If successful, PlannedAction is exposed in the ViewModel.
	•	Status Indicators: StatusEffectService returns a list of formatted tags ([BLOOD 3]) for each combatant.
2. Logic Decision Trees
Intent Reveal Logic:
	1	Check: Is Player Class == Jötun-Reader OR Analysis Ability Active?
	◦	Yes: Reveal All.
	2	Else Check: Player WITS + Perception > Enemy Stealth?
	◦	Yes: Reveal Action Type (e.g., "Attacking").
	◦	No: Show "?".
	3	Icon Mapping:
	◦	Attack -> ⚔️
	◦	Defend -> 🛡️
	◦	Buff/Debuff -> ✨
	◦	Ultimate -> ☠️
