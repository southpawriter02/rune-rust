Here is the comprehensive implementation plan for v0.2.0a: The Arena (State & Initiative).
v0.2.0a: The Arena
Goal: Establish the underlying infrastructure for the Combat System. This includes the CombatState machine, the Combatant adapter pattern, and the Initiative sorting logic. By the end of this phase, the game will be able to transition from Exploration to Combat, roll initiative, cycle through turns, and return to Exploration.

1. Implementation Workflow
	1	Domain Entities: Create CombatState, Combatant, and Enemy (stub) in Core.
	2	State Management: Update GameState to hold the active CombatState.
	3	Initiative Logic: Implement InitiativeService using IDiceService (d10 + FINESSE + WITS).
	4	Orchestration: Implement CombatService to handle StartCombat, NextTurn, and EndCombat.
	5	Integration: Wire debug-combat command in CommandParser to test transitions.
	6	Verification: Unit tests for sorting logic; Integration tests for state transitions.

using RuneAndRust.Core.Entities;
using RuneAndRust.Core.Enums;
using Attribute = RuneAndRust.Core.Enums.Attribute;

namespace RuneAndRust.Core.Models.Combat;

public class Combatant
{
    public Guid Id { get; set; } = Guid.NewGuid();
    public string Name { get; set; } = string.Empty;
    public bool IsPlayer { get; set; }

    // Volatile Combat State
    public int Initiative { get; set; }
    public int CurrentHp { get; set; }
    public int MaxHp { get; set; }

    // Source References
    public Character? CharacterSource { get; set; }
    public Enemy? EnemySource { get; set; }

    public int GetAttribute(Attribute attr)
    {
        if (CharacterSource != null) return CharacterSource.GetAttribute(attr);
        if (EnemySource != null) return EnemySource.Attributes.GetValueOrDefault(attr, 0);
        return 0;
    }

    public static Combatant FromCharacter(Character c) => new()
    {
        Name = c.Name,
        IsPlayer = true,
        CharacterSource = c,
        CurrentHp = c.CurrentHp,
        MaxHp = c.MaxHp
    };

    public static Combatant FromEnemy(Enemy e) => new()
    {
        Name = e.Name,
        IsPlayer = false,
        EnemySource = e,
        CurrentHp = e.MaxHp, // Start full
        MaxHp = e.MaxHp
    };
}

B. Engine Layer (Services)
File: RuneAndRust.Engine/Services/InitiativeService.cs
using Microsoft.Extensions.Logging;
using RuneAndRust.Core.Interfaces;
using RuneAndRust.Core.Models.Combat;
using Attribute = RuneAndRust.Core.Enums.Attribute;

namespace RuneAndRust.Engine.Services;

public class InitiativeService
{
    private readonly IDiceService _dice;
    private readonly ILogger<InitiativeService> _logger;

    public InitiativeService(IDiceService dice, ILogger<InitiativeService> logger)
    {
        _dice = dice;
        _logger = logger;
    }

    public void RollInitiative(Combatant combatant)
    {
        // Formula: d10 + FINESSE + WITS (Vigilance)
        var roll = _dice.Roll(1, "Initiative").Successes; // Temporary logic until full dice mechanics
        // NOTE: Real implementation might use raw die value, not successes.
        // For v0.2.0a, we assume 1d10 value is passed differently or just use successes.
        // Let's assume DiceService has a RollOne() or we take the sum.

        var vigilance = combatant.GetAttribute(Attribute.Finesse) + combatant.GetAttribute(Attribute.Wits);
        combatant.Initiative = roll + vigilance;

        _logger.LogDebug("{Name} rolled initiative: {Total} (Vigilance {Vig} + Roll {Roll})",
            combatant.Name, combatant.Initiative, vigilance, roll);
    }

    public List<Combatant> SortTurnOrder(IEnumerable<Combatant> combatants)
    {
        return combatants
            .OrderByDescending(c => c.Initiative)
            .ThenByDescending(c => c.GetAttribute(Attribute.Finesse))
            .ThenBy(_ => Guid.NewGuid()) // Chaos tie-breaker
            .ToList();
    }
}
File: RuneAndRust.Engine/Services/CombatService.cs
using Microsoft.Extensions.Logging;
using RuneAndRust.Core.Entities;
using RuneAndRust.Core.Enums;
using RuneAndRust.Core.Models;
using RuneAndRust.Core.Models.Combat;

namespace RuneAndRust.Engine.Services;

public class CombatService
{
    private readonly GameState _gameState;
    private readonly InitiativeService _initiative;
    private readonly ILogger<CombatService> _logger;

    public CombatService(GameState gameState, InitiativeService initiative, ILogger<CombatService> logger)
    {
        _gameState = gameState;
        _initiative = initiative;
        _logger = logger;
    }

    public void StartCombat(List<Enemy> enemies)
    {
        if (_gameState.CurrentCharacter == null) return;

        _logger.LogInformation("Initializing Combat Encounter...");
        var state = new CombatState();

        // 1. Add Player
        var player = Combatant.FromCharacter(_gameState.CurrentCharacter);
        _initiative.RollInitiative(player);
        state.TurnOrder.Add(player);

        // 2. Add Enemies
        foreach (var enemy in enemies)
        {
            var c = Combatant.FromEnemy(enemy);
            _initiative.RollInitiative(c);
            state.TurnOrder.Add(c);
        }

        // 3. Sort
        state.TurnOrder = _initiative.SortTurnOrder(state.TurnOrder);

        // 4. Update State
        _gameState.CombatState = state;
        _gameState.Phase = GamePhase.Combat;

        _logger.LogInformation("Combat Started. Round 1. Active: {Name}", state.ActiveCombatant.Name);
    }

    public void NextTurn()
    {
        if (_gameState.CombatState == null) return;
        var state = _gameState.CombatState;

        state.TurnIndex++;
        if (state.TurnIndex >= state.TurnOrder.Count)
        {
            state.TurnIndex = 0;
            state.RoundNumber++;
            _logger.LogInformation("Round {Round} Begins.", state.RoundNumber);
        }

        _logger.LogInformation("Turn: {Name}", state.ActiveCombatant.Name);
    }

    public void EndCombat()
    {
        _gameState.Phase = GamePhase.Exploration;
        _gameState.CombatState = null;
        _logger.LogInformation("Combat Ended.");
    }
}

4. Logic Decision Tree
	•	Q: How do we handle "Surprise"?
	◦	A: Not in v0.2.0a. Add a bool isSurprise flag to StartCombat later. For now, everyone rolls standard initiative.
	•	Q: Do attributes update mid-combat?
	◦	A: Combatant keeps a reference to the source (Character). If Character attributes change (e.g. buff), GetAttributereflects it immediately. HP is copied to Combatant.CurrentHp so combat damage doesn't immediately kill the persistent entity until resolution.
	•	Q: What if the DiceService only returns successes (0-10)?
	◦	A: For Initiative, we need a raw 1-10 value. Update IDiceService or use Random.Shared inside InitiativeService if IDiceService is strictly pool-based. (Recommended: Add RollDie(int sides) to IDiceService).
