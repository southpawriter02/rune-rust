v0.2.1a: The Gear (Equipment Integration)
Goal: Transition combat math from "Debug Mode" (fixed d6 damage, 0 soak) to "Live Mode" (variable damage based on weapons, variable soak based on armor).
1. Architecture & Data Flow
We effectively "snapshot" the character's equipment state when combat begins to avoid querying the database every time an attack is rolled.
Workflow Diagram:
	1	Combat Start: CombatService calls Combatant.FromCharacter(character).
	2	Snapshot: Combatant iterates through character.Inventory (where IsEquipped == true).
	3	Calculation:
	◦	Weapon: Identify Main Hand / Two-Handed item. Cache DamageDie.
	◦	Armor: Sum SoakBonus from all armor slots. Cache ArmorSoak.
	4	Execution: AttackResolutionService uses cached Combatant stats for resolution.
	5	Victory: CombatService triggers LootService and returns a CombatResult object.

2. Logic Decision Trees
A. Weapon Selection Logic
Input: Character Inventory (Equipped)
	1	Check Two-Handed Slot: Is there an item?
	◦	Yes: Use this Item. Stop.
	2	Check Main Hand Slot: Is there an item?
	◦	Yes: Use this Item. Stop.
	3	Fallback: Use "Unarmed" (d4 Damage, Physical).
B. Armor Calculation Logic
Input: Character Inventory (Equipped)
	1	Initialize: TotalSoak = 0.
	2	Iterate: For each equipped item:
	◦	Is item Equipment?
	◦	Does it have SoakBonus > 0?
	◦	Yes: TotalSoak += SoakBonus.
	3	Result: Cache TotalSoak on Combatant.

using RuneAndRust.Core.Entities;
using RuneAndRust.Core.Enums;

public partial class Combatant
{
    // New Cached Stats
    public int WeaponDamageDie { get; set; } = 4; // Default Unarmed (d4)
    public int WeaponAccuracyBonus { get; set; } = 0;
    public int ArmorSoak { get; set; } = 0;
    public string WeaponName { get; set; } = "Fists";

    public static Combatant FromCharacter(Character c)
    {
        var combatant = new Combatant
        {
            Id = Guid.NewGuid(),
            Name = c.Name,
            IsPlayer = true,
            CurrentHp = c.CurrentHp,
            MaxHp = c.MaxHp,
            // ... (Initiative calculated separately)
        };

        // Snapshot Equipment Stats
        var equippedItems = c.Inventory.Where(i => i.IsEquipped).Select(i => i.Item).OfType<Equipment>();

        // 1. Calculate Soak
        combatant.ArmorSoak = equippedItems.Sum(e => e.SoakBonus);

        // 2. Determine Weapon (Priority: 2H -> MainHand -> Unarmed)
        var weapon = equippedItems.FirstOrDefault(e => e.Slot == EquipmentSlot.TwoHand)
                  ?? equippedItems.FirstOrDefault(e => e.Slot == EquipmentSlot.MainHand);

        if (weapon != null)
        {
            combatant.WeaponDamageDie = weapon.DamageDie;
            combatant.WeaponName = weapon.Name;
            // Future: Add AttributeBonuses handling here
        }

        return combatant;
    }
}

public AttackResult ResolveMeleeAttack(Combatant attacker, Combatant defender, AttackType type)
{
    // 1. Determine Dice
    // Use cached weapon die instead of static lookups
    // AttackType modifiers (Light/Heavy) might modify the base weapon die or add flat damage
    int weaponDie = attacker.WeaponDamageDie;

    // ... Roll to Hit Logic (Success vs Defense) ...

    // 2. Calculate Damage
    // Roll the weapon die
    int weaponRoll = _dice.RollSingle(weaponDie, "Weapon Damage");
    int might = attacker.GetAttribute(Attribute.Might);

    int rawDamage = weaponRoll + might;

    // 3. Apply Mitigation
    int soak = defender.ArmorSoak; // Use cached soak
    int finalDamage = Math.Max(1, rawDamage - soak); // Minimum 1 rule

    return new AttackResult(outcome, netSuccesses, rawDamage, finalDamage, true);
}
